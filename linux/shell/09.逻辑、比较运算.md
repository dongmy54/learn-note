#### 逻辑、比较运算
> 1. test
> 2. &&、|| 、！

#### 用法
> - test <=> []
> - 扩展成[[]] 可以支持&&、||
> - PS: [] 里面内容需要和方括号保持空格

```bash
#---------------字符----------------
# z 即 zero 判断字符是否为空
[ -z '' ] 
echo $?
# => 0
[ -z "sd" ]
echo $?
# => 1

# 判断相同
[ 'sd' = 'sd' ]
echo $?
# => 0

# 判断不同
[ 'ad' != 'd' ]
echo $?
# => 0


#---------------整数-------------------
# >
[ 5 -gt 3 ]  # <=> (( 5 > 3 ))
echo $?
# => 0

# >=
[ 5 -ge 4 ]  # <=> (( 5 >= 4 ))
echo $?
# => 0

# <
[ 5 -lt 8 ]  # <=> ((5 < 8))
echo $?
# => 0

# <=
[ 5 -le 8 ]  # <=> ((5 <= 8))
echo $?
# => 0

# ==
[ 5 -eq 5 ]  # <=> ((5 == 8))
echo $?
# => 0


#---------------文件-------------------
# -d (directroy) 是否存在此目录
[ -d /Users ]
echo $?
# => 0

# -e (exits) 目录/文件是否存在
[ -e /Users/dongmingyan ]
echo $?
# => 0

# -f (file) 文件是否存在
[ -f /Users/dongmingyan/shell/test.sh ]
echo $?
# => 0


#--------------&&、||、!----------------
[[ 5 -gt 4 && 6 -lt 4 ]]
echo $?
# => 1

[[ 5 -gt 4 || 6 -lt 4 ]]
echo $?
# => 0

[[ 5 -gt 4 && ! 6 -lt 4 ]]
echo $?
# => 0
```
