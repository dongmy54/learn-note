#### setInterval
> 间隔执行
```js
window.setInterval(function(){
  str_time_desc = timeStampToCountDown(end_time_stamp);

  if (str_time_desc === 'end_of_countdown'){
    current_obj.text('逾期未确认');
  } else {
    current_obj.text(str_time_desc);
  };
}, 1000);
```
